확장

  • icon 추가, label 추가
  • 레이블을 추가할 경우, 레이블을 먼저 선언해준다.
gnui-guide.png 
<!-- 아이콘만 사용할 경우 -->
<label class="gn-file">
  <input type="file">
  <span class="gn-icon is-small">
    <i class="fas fa-upload"></i>
  </span>
</label>

<!-- 텍스트와 동일한 색깔의 아이콘 사용할 경우 -->
<label class="gn-file is-info is-outline">
  <input type="file">
  Select File
  <span class="gn-icon is-small">
    <i class="fas fa-angle-right"></i>
  </span>
</label>

<!-- 텍스트와 다른 색깔의 아이콘 사용할 경우 -->

  <span class="gn-icon is-small is-black is-right">
    <i class="fas fa-angle-right"></i>
  </span>
  <label class="gn-file is-info">

    Select File
  </label>
</div>
