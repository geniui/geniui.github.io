  Gn.create('calendar', {
    target: '.new-calendar',
    value: '2021-04-01',
    onSelect: function (v) {
      console.log('선택한 날짜: '+ v)
    },
    max: '2021-04-30',
    min: '2021-03-01'
  });
  Gn.create('calendar', {
    target: '.new-calendar1',
    onSelect: function (v) {
      console.log('선택한 날짜: '+ v)
    },
    dateType: 'datetime',
    max: '1M',
    min: '-1M'
  });
name type default description
target string calendar 생성 타겟
value string | Date now 초기 날짜
dateType string date date, datetime
onSelect function(value) 날짜 선택 시 발생 이벤트

name return description
